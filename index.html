function reservar(){ const docente=document.getElementById("docente").value; if(!docente) return alert("Selecciona tutor"); if(seleccionados.length===0) return alert("Selecciona horario"); const { jsPDF } = window.jspdf; // 🔒 VERIFICAR SI YA RESERVÓ db.ref("reservas").once("value", snapshot => { let yaReservo = false; snapshot.forEach(child=>{ if(child.val().nombre === docente){ yaReservo = true; } }); if(yaReservo){ alert("Este tutor ya realizó una reserva"); return; } // ✅ SI NO HA RESERVADO seleccionados.forEach(slot=>{ const id = slot.dataset.id; db.ref("reservas/"+id).once("value", snap=>{ if(snap.exists()){ alert("Ese horario ya fue tomado"); return; } const fecha = new Date().toLocaleString(); db.ref("reservas/"+id).set({ nombre: docente, fecha: fecha }); // 📄 BOLETA const doc = new jsPDF(); doc.text("BOLETA DE INGRESO AIP", 20, 20); doc.text("Tutor: " + docente, 20, 40); doc.text("Horario: " + id, 20, 50); doc.text("Fecha: " + fecha, 20, 60); doc.text("PIP: YARVIN RS", 20, 80); doc.save("boleta_AIP.pdf"); }); }); alert("Reserva confirmada"); seleccionados=[]; }); }
